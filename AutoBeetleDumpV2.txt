while not @findobject masterrunebook
  headmsg 'Select main runebook..' 46
  promptalias masterrunebook
endwhile
while not @findalias beetle
  if mounted self
    @useobject self
    pause 800
  endif
  headmsg 'Select your beetle..' 46
  promptalias beetle
endwhile
if not mounted self
  @useobject beetle
  pause 1000
endif
headmsg 'Recalling to Bank..' 46
@useobject masterrunebook
waitforgump 0x554b87f3 4000
replygump 0x554b87f3 5
pause 3000
msg 'bank'
pause 1000
if mounted self
  headmsg 'Dismount' 46
  @useobject self
  pause 1000
endif
headmsg 'Opening Beetle' 46
waitforcontext beetle 10 15000
pause 1000
while @findtype 0x1081 any beetle
  headmsg 'Moving Leather' 46
  @moveitem found bank
  pause 1200
endwhile
while @findtype 0x26b4 any beetle
  headmsg 'Moving Scales' 46
  moveitem found bank
  pause 1200
endwhile
if not mounted self
  headmsg 'Mount' 46
  @useobject beetle
  pause 1000
endif
pause 1000
headmsg 'Recalling to farm..' 46
@useobject masterrunebook
waitforgump 0x554b87f3 4000
replygump 0x554b87f3 11
